UPF对接FRR之OSPF

UPF

前言

接续上一篇”UPF对接FRR之BGP”，这里简单列一下OSPF的配置。

一、主用UPF的OSPF配置参考

主用UPF上的配置如下:

[root@testdept-upf-32-gsu-0 ~]# cat /usr/local/frr/etc/ospfd.conf
interface vEth0
 ip ospf bfd
 ip ospf bfd profile bfd
!
interface vEth1
 ip ospf bfd
 ip ospf bfd profile bfd
!
router ospf
 ospf router-id 172.19.63.201
 network 172.19.61.25/32 area 1
 network 172.19.62.25/32 area 2
 network 172.19.63.26/32 area 1
 network 172.19.64.26/32 area 2
 network 190.0.0.0/8 area 2
 neighbor 172.19.61.254
 neighbor 172.19.62.254

 router-info area
!
ipv6 access-list accarea1 seq 5 permit 172:19:0:63::/64
ipv6 access-list accarea2 seq 5 permit 172:19:0:64::/64
ipv6 access-list accarea2 seq 10 permit 190::/16
!
!
line vty
!

[root@testdept-upf-32-gsu-0 ~]# cat /usr/local/frr/etc/ospf6d.conf
interface eth3
 ipv6 ospf6 area 0
!
interface vEth0
 ipv6 ospf6 area 1
 ipv6 ospf6 bfd
 ipv6 ospf6 bfd profile bfd
!
interface vEth1
 ipv6 ospf6 area 2
 ipv6 ospf6 bfd
 ipv6 ospf6 bfd profile bfd
!
router ospf6
 ospf6 router-id 172.19.63.200
 area 1 range 172:19:0:63::26/128
 area 1 export-list accarea1
 area 2 range 172:19:0:64::26/128
 area 2 range 190::/16
 area 2 export-list accarea2
!
ipv6 access-list accarea1 seq 5 permit 172:19:0:63::/64
ipv6 access-list accarea2 seq 5 permit 172:19:0:64::/64
ipv6 access-list accarea2 seq 10 permit 190::/16
!
!
line vty
!

二、备用UPF的OSPF配置参考

备用UPF上的配置如下:

[root@testdept-upf-32-gsu-1 ~]# cat /usr/local/frr/etc/ospfd.conf
interface vEth0
 ip ospf bfd
 ip ospf bfd profile bfd
!
interface vEth1
 ip ospf bfd
 ip ospf bfd profile bfd
!
router ospf
 ospf router-id 172.19.63.200
 network 172.19.61.24/32 area 1
 network 172.19.62.24/32 area 2
 neighbor 172.19.61.254
 neighbor 172.19.62.254

 router-info area
!
ipv6 access-list accarea1 seq 5 permit 172:19:0:63::/64
ipv6 access-list accarea2 seq 5 permit 172:19:0:64::/64
ipv6 access-list accarea2 seq 10 permit 190::/16
!
!
line vty
!

[root@testdept-upf-32-gsu-1 ~]# cat /usr/local/frr/etc/ospf6d.conf
interface eth3
 ipv6 ospf6 area 0
!
interface vEth0
 ipv6 ospf6 area 1
 ipv6 ospf6 passive
 ipv6 ospf6 bfd
 ipv6 ospf6 bfd profile bfd
!
interface vEth1
 ipv6 ospf6 area 2
 ipv6 ospf6 passive
 ipv6 ospf6 bfd
 ipv6 ospf6 bfd profile bfd
!
router ospf6
 ospf6 router-id 172.19.63.201
 area 1 export-list accarea1
 area 2 export-list accarea2
!
ipv6 access-list accarea1 seq 5 permit 172:19:0:63::/64
ipv6 access-list accarea2 seq 5 permit 172:19:0:64::/64
ipv6 access-list accarea2 seq 10 permit 190::/16
!
!
line vty
!
